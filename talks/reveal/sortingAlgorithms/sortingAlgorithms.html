          Function insertionSort(A : Array of Element; n : N)
            for i := 2 to n do
              j := i
              while (j > 0) & (A[j − 1] > A[j])
                swap(A[j − 1], A[j])
                j := j − 1
              end while
            invariant A[1] ≤ · · · ≤ A[i]
          end for

          Function selectionSort(A : Array of Element; n : N)
          for i := 1 to n do
            min := i
            for j := i+1 to n do
              if A[j] < A[min] then
                min := j
            endfor

          Function quickSort(A : Array of Element; l, r : N)
            if l ≥ r then
              return
            q := pickPivotPos(A, l, r)
            m := partition(A, l, r, q)
            quickSort(A, l, m − 1), quickSort(A, m + 1, r)
